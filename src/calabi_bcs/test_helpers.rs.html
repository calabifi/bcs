6/// Asserts that a value can be serialized and deserialized back to an equal value.
7///
8/// This is a test helper function that verifies round-trip serialization.
9///
10/// # Panics
11///
12/// Panics if:
13/// - Serialization fails
14/// - Deserialization fails
15/// - The deserialized value is not equal to the original
16pub fn assert_canonical_encode_decode<T>(t: &T)
17where
18    T: crate::TBCSSerialize + crate::TBCSDeserializeOwned + std::fmt::Debug + PartialEq,
19{
20    let bytes = crate::to_bytes(t).unwrap();
21    let s: T = crate::from_bytes(&bytes).unwrap();
22    assert_eq!(*t, s);
23}
